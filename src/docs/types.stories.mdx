# Exposed types

**DeepPartial** 
``` ts
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends Array<infer U>
  ? Array<DeepPartial<U>>
  : T[P] extends ReadonlyArray<infer U>
  ? ReadonlyArray<DeepPartial<U>>
  : DeepPartial<T[P]>
};

```
**IGlobalConfig** - global config props

``` ts
interface IGlobalConfig {
  locale: Locale;
  textOverrides: Partial<TTexts>;
}
```

``` ts
interface INumberFormat {
```

``` ts
interface IValidationTargetProps {
```

**IMenuItem**

``` ts
interface IMenuItem {
  label: ReactNode;
  value: string;
  onClick?: (value: string) => void;
}
```

**IPageHeaderAction**

``` ts
interface IPageHeaderAction extends Pick<IButtonProps, 'className' | 'color' | 'disabled' | 'variant' | 'loading' | 'menuItems'>, Omit<IMenuItem, 'onClick' | 'value'> {
  value?: string;
  Icon: ComponentType<SvgIconProps>;
  onClick?: (value?: string) => void;
  type?: 'button' | 'icon';
}
```

**IPageHeaderTab**

``` ts
interface IPageHeaderTab extends Omit<IMenuItem, 'onClick'> {}
```

**IOption**

``` ts
interface IOption<TValue extends string | number = string | number> {
  label: string;
  value: TValue;
}
```

**IDisableableOption** - extends `IOption` with optional `disabled` field.

``` ts
interface IDisableableOption<TValue extends string | number = string | number> extends IOption<TValue> {
  disabled?: boolean;
}
```

**IListItem**

``` ts
interface IListItem {
```

**ISortingItem** - extends `react-sortablejs`'s `ItemInterface` that has `id` as a required field.

```ts
interface ISortingItem extends ItemInterface, IListItem { }
```

**INavigationMenuItem**

```ts
interface INavigationMenuItem<MenuItemType extends IMenuItem = IMenuItem> extends IMenuItem, Pick<IButtonProps<MenuItemType>, 'disabled' | 'tabIndex' | 'className' | 'innerRef' | 'type' | 'MenuItem' | 'menuItems' | 'getMenuItemSelected'> {
  Icon: ComponentType<SvgIconProps>;
}
```

**INavigationMenuItemsGroup**

```ts
interface INavigationMenuItemsGroup {

  Icon: ComponentType<SvgIconProps>;

```

**TNavigationMenuItem**

```ts
type TNavigationMenuItem = INavigationMenuItem | INavigationMenuItemsGroup
```
